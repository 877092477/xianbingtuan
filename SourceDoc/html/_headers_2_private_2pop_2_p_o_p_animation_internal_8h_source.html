114 struct ComputeProgressFunctor<Vector4r> {
115  CGFloat operator()(const Vector4r &value, const Vector4r &start, const Vector4r &end) const {
116  CGFloat s = (value - start).squaredNorm(); // distance from start
117  CGFloat e = (value - end).squaredNorm(); // distance from end
118  CGFloat d = (end - start).squaredNorm(); // distance from start to end
119 
120  if (0 == d) {
121  return 1;
122  } else if (s > e) {
123  // s -------- p ---- e OR s ------- e ---- p
124  return sqrtr(s/d);
125  } else {
126  // s --- p --------- e OR p ---- s ------- e
127  return 1 - sqrtr(e/d);
128  }
